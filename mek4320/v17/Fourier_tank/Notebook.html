g = sympy.symbols('g', real=True)
case = {L: sympy.S(10),
        f: sympy.S(1),
        A: sympy.S(1)/5,
        g: sympy.S(981)/100
        }
L2 = case[L]/2
nmax = 10
display(case)

a0_case = sympy.integrate(init/L, (x, -f, f)).subs(case)

# Collect functions to plot
init2 = sympy.Piecewise((0, x < -f), (0, x > f), (init, True))
display(init2)
funcs = [(sympy.lambdify(x, init2.subs(case)), 'init')]
for t in (0, 0.5, 1, 1.5, 2):
    k = 2*pi*n/L
    w = sympy.sqrt(k*g)
    core2 = Is*cos(k*x)*cos(w*t)
    core2_case = core2.subs(case)
    eta_nmax = a0_case + sympy.summation(core2_case, (n, 1, nmax))
    
    funcs.append((sympy.lambdify(x, eta_nmax), 't=%r' % t))

# Plot it
pyplot.figure(figsize=(14,8))
xvec = numpy.linspace(-L2, L2, 101)
for func, name in funcs:
    pyplot.plot(xvec, [func(xi) for xi in xvec], label=name)
pyplot.legend();
